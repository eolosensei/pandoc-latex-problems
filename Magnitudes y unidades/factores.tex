

\begin{exercise}[
    tags    = {unidades, factores de conversión, conversión},
    topics  = {unidades, medida, física, química},
    source  = {MERES},
  ]
  Utiliza factores de conversión para realizar los siguientes cambios de unidades:
  \begin{multicols}{2}
    \begin{enumerate}
      \item \SI{324500}{\mg} a \si{\mega\g}
      \item \SI{3}{\centi\g} a \si{\giga\g}
      \item \SI{90}{\dm} a \si{\nm}
      \item \SI{86400}{\ms} a \si{\hour}
      \item \SI{0.35}{\hL} a \si{\uL}
      \item \SI{0.335}{\mL} a \si{\uL}
      \item \SI{0.00092}{\hecto\m} a \si{\pm}
      \item \SI{0.075}{\hecto\g} a \si{\ng}
    \end{enumerate}
  \end{multicols}
\end{exercise}

\begin{solution}
  \begin{enumerate}
    \item \geconvertbigger{324500}{\mg}{e9}{\mega\g}{3.245e-4}
    \item \geconvertbigger{3}{\centi\g}{e11}{\giga\g}{3e-11}
    \item \geconvertlesser{90}{\dm}{e8}{\nm}{9e9}
    \item \geconvertbigger{86400}{\ms}{3600e3}{\hour}{2.4e-2}
    \item \geconvertlesser{0.35}{\hL}{e8}{\uL}{3.5e7}
    \item \geconvertlesser{0.355}{\mL}{e3}{\uL}{3.35e2}
    \item \geconvertlesser{0.0092}{\hecto\m}{e14}{\pm}{9.2e10}
    \item \geconvertlesser{0.075}{\hecto\g}{e11}{\ng}{7.5e9}
  \end{enumerate}
\end{solution}
